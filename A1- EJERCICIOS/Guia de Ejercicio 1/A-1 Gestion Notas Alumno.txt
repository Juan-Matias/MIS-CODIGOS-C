int obtenerCantidadNotas(float  aNotas[DA]){
	int c;
	int cantNotas=0;
	for(c=0;c<DA;c++){
		if(aNotas[c]>0.0F){
			cantNotas++;
		}
	}
	return cantNotas;
}
float obtenerPromedio(float aNotas[]){
	int c;
	float sumaNotas,promNotas;
	sumaNotas=0.0F;
	for(c=0;c<DA;c++){
		if(aNotas[c]>0.0F){
			sumaNotas += aNotas[c];
		}
	}
	promNotas=sumaNotas / CANTIDAD_NOTAS;
	return promNotas;	
}
